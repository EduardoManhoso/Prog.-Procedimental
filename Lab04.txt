//Exercício 16
#include <stdio.h>

main(){
    int op, num1, num2;
    do{
        do{
            printf("1. Adicao\n");
            printf("2. Subtracao\n");
            printf("3. Multiplicacao\n");
            printf("4. Divisao\n");
            printf("5. Sair\n");
            printf("Opcao: ");
            scanf("%d", &op);
        }while(op < 1 || op > 5);
        if(op != 5){
            printf("Digite dois numeros inteiros: ");
            scanf("%d %d", &num1, &num2);
            switch(op){
                case 1:
                    printf("%d + %d = %d\n", num1, num2, num1 + num2);
                    break;
                case 2:
                    printf("%d - %d = %d\n", num1, num2, num1 - num2);
                    break;
                case 3:
                    printf("%d * %d = %d\n", num1, num2, num1 * num2);
                    break;
                case 4:
                    printf("%d / %d = %.2f\n", num1, num2, (double) num1 / num2);
                    break;
            }
        }
    }while(op != 5);
}
