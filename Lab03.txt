#include <math.h>

main()
{   
    int a, b, c, delta;
    double x1, x2;
    printf("Digite o valor das coeficientes na ordem ax^2 + bx + c : ");
    scanf("%d %d %d", &a, &b, &c);
    if(a == 0)
        printf("Nao e equacao do segundo grau.");
    else{
        delta = pow(b, 2) - (4 * a * c);
        if(delta > 0){
            x1 = (-b + sqrt(delta)) / (2 * a);
            x2 = (-b - sqrt(delta)) / (2 * a);
            printf("Existe dua raizes, %lf e %lf", x1, x2);
        }else if(delta < 0)
            printf("Nao existe raizes.");
        else{
            x1 = (-b + sqrt(delta)) / (2 * a);
            printf("Existe so uma raiz, %lf", x1);
        }
    }
}
